map()

var numbers = [1, 2, 3, 2, 1];
var result = numbers.map(function(item, index, array) {
    return item * 2;
});
console.log(result);   // [2, 4, 6, 4, 2]

map() + arrow function

var numbers = [1, 2, 3, 2, 1];
let result = numbers.map(item => {
  return item*3;
})      
console.log(result);   // [2, 4, 6, 4, 2]
